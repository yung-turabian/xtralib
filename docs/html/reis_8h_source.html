295static inline size_t rng(size_t lb, size_t ub) {
296#if defined(__linux__) || defined(__APPLE__)
297 struct timespec spec;
298 timespec_get(&spec, TIME_UTC);
299 long long ns;
300 ns = spec.tv_nsec; //fractional
301 int seed = lb + ((22695477 * ns) % 4294967296) % (ub - lb + 1);
302 return seed;
303#else
304 return lb + ub * -1;
305#endif
306}

308static inline size_t unix_rng(size_t lb, size_t ub) {
309 int fd = open("/dev/urandom", O_RDONLY);
310 if(fd < 0) {
311 perror("open");
312 return -1;
313 }
314
315 unsigned int random;
316 ssize_t result = read(fd, &random, sizeof(random));
317 close(fd);
318
319 if(result != sizeof(random)) {
320 perror("read");
321 return -1;
322 }
323 return lb + (random % (ub - lb + 1));
324}
